Run Jupyter Notebook within Virtualenv

2018-04-28

# create virtualenv myenv
mkvirtualenv myenv

# enter virtualenv
workon myenv

# install jupyter if it's not done yet
(myenv) pip3 install jupyter

# install ipykernel if it's not done yet
(myenv) pip3 install ipykernel

# create new kernel of python3
(myenv) python3 -m ipykernel install --user --name myenv --display-name "Python 3"

# start the notebook within myenv
(myenv) jupyter notebook
